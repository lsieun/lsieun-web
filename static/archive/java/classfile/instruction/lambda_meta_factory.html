Runnable

import java.lang.invoke.*; public class HelloWorld { public static void main(String[] args) throws Throwable { // 第一步，FunctionalInterface的信息 MethodType invoked_class_type = MethodType.methodType(Runnable.class); String invoked_method_name = "run"; MethodType invoked_method_type = MethodType.methodType(void.class); // 第二步，寻找一个具体的实现方法 MethodHandles.Lookup caller = MethodHandles.lookup(); System.out.println("caller: " + caller); MethodType target_method_type = MethodType.methodType(void.class); MethodHandle target_method_handle = caller.findStatic(HelloWorld.class, "print", target_method_type); System.out.println("target method handle: " + target_method_handle); // 第三步，生成CallSite对象（形象的来说，我们组装了一件武器） MethodType instantiated_method_type = MethodType.methodType(void.class); CallSite site = LambdaMetafactory.metafactory( caller, invoked_method_name, invoked_class_type, invoked_method_type, target_method_handle, instantiated_method_type); // 第四步，以FunctionalInterface的视角，来调用具体的实现方法 MethodHandle factory = site.getTarget(); System.out.println("factory method handle: " + factory); Runnable r = (Runnable) factory.invoke(); r.run(); } private static void print() { System.out.println("hello world"); } }

运行GoodBoy.java的main方法，会出现错误。出错误的原因：我们试图在GoodBoy.java中去访问HelloWorld.java中的私有（private）方法print。如果我们将HelloWorld.java中的print方法修改为public，就不会出现错误了。

生成两个bootstrap方法

在下面的代码中，我们使用了两个lambda表达式，因此在BootstrapMethods属性中会有两个bootstrap method。

import java.util.Random; import java.util.function.BiFunction; public class HelloWorld { public static int add(int a, int b) { return a + b; } public static int sub(int a, int b) { return a - b; } public static void test(BiFunction<Integer, Integer, Integer> func, int a, int b) { System.out.println(func.getClass()); Integer result = func.apply(a, b); System.out.println(result); } public static void main(String[] args) { Random rand = new Random(); boolean flag = rand.nextBoolean(); HelloWorld.test(flag ? HelloWorld::add : HelloWorld::sub, 20, 5); } }
