Array Class

For an array of references, the array class is marked as having been defined by the defining class loader of the component type.很重要
For an array of primitive types, the array class is marked as having been defined by the bootstrap class loader.很重要

package sample;

public class HelloWorld {
}

package sample;

import com.sun.javafx.util.Logging;

public class Program {
    public static void main(String[] args) {
        // Bootstrap Class Loader
        printClassLoader(int.class);
        printClassLoader(int[].class); // array of primitive types
        printClassLoader(String.class);
        printClassLoader(String[].class); // array of references


        // Extension Class Loader
        printClassLoader(Logging.class);
        printClassLoader(Logging[].class);

        // Application Class Loader
        printClassLoader(HelloWorld.class);
        printClassLoader(HelloWorld[].class);
    }

    private static void printClassLoader(Class clazz) {
        ClassLoader classLoader = clazz.getClassLoader();
        System.out.println(classLoader);
    }
}
