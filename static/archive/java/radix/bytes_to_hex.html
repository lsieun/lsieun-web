Formatter

public static String toHex(byte[] bytes) { StringBuilder sb = new StringBuilder(); Formatter fm = new Formatter(sb); for (byte b : bytes) { fm.format("%02X", (b & 0xFF)); } return sb.toString(); }

Integer.toString

public static String toHex(byte[] bytes) { StringBuilder sb = new StringBuilder(); for (int i = 0; i < bytes.length; ++i) { sb.append(Integer.toString((bytes[i] & 0xFF) + 0x100, 16).substring(1)); } return sb.toString(); }
