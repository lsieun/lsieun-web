public class IntegerUtils { public static String toBinary(int num) { return toBinary(num, Integer.SIZE); } public static String toBinary(int num, int size) { StringBuilder sb = new StringBuilder(); for (int i = size - 1; i >= 0; i--) { sb.append((num >> i) & 0x01); } return sb.toString(); } public static void main(String[] args) { int val = 0x100; String str = toBinary(val); System.out.println(str); } }
